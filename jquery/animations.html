<a href="#" class="btn btn-md btn-default start">Begin animation</a>
<div class="animate-me"></div>

.animate-me {
background: #374252;
height: 100px;
position: relative;
width: 100px;	
}

$(document).ready(function(){
    $("a.start").click(function(){
        $(".animate-me").animate({
            left: '40%',
            opacity: '0.5',
            height: '150px',
			top: '0px',
            width: '150px'
        });
    });
});
